pipeline  {
  agent any
  stages {
    stage ("clean") {
    steps {
      echo "cleaning"
      bat 'mvn clean'
    }
  }
  stage ("packing") {
        steps {
          echo "packing"
          bat 'mvn package'
        }
  }
    stage ("build") {
      steps {
        echo "building"
        bat 'mvn install'
            }
}

    stage ("deploy") {
      steps {
          echo "deploying"
          deploy adapters: [tomcat9(alternativeDeploymentContext: '', credentialsId: 'tomcat', path: '', url: 'http://localhost:8080/')], contextPath: null, war: '**/*.war'
      }
    }
  }
}
