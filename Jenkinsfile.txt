pipeline  {
  agent any
  stages {
    stage ("clean") {
    steps {
      echo "cleaning"
      bat 'mvn clean'
    }
  }
  stage ("packing") {
        steps {
          echo "packing"
          bat 'mvn package'
        }
  }
    stage ("build") {
      steps {
        echo "building"
        bat 'mvn install'
            }
}
    stage ("Deploy") {
      steps {
        echo "deploying"
        bat 'mvn deploy'
       }    
    }
  }
}
